Tiffany's Pong

This project demostrates my object-oriented programming (OOP) skills using C#.
It is a common game I developed using Unity called "Tiffany's Pong".
Below you will see a snippet of code written by me as well as a button that allows you to play the game.
Here are the controls to play.

Controls

'w' to move up
's' to move down
'd' to respawn pong
'r' to restart

Code Snippet
(Spawn Pong)

/*Created by Jerome Johnson*/

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

public class SpawnPong : MonoBehaviour {

 public GameObject pong;
 public Vector3 spawnValues;
 public int pongCount;
 public float spawnWait;
 public float startWait;
 public Text winText;

 void Start(){
   //winText.text = "";
   //restartText.text = "";
   StartCoroutine (Spawning ());
 }

 void Update(){
   if (Input.GetKeyDown (KeyCode.D)) {
    Destroy (GameObject.FindGameObjectWithTag ("Pong Ball"));
    StartCoroutine (Spawning ());
   }
 }

 IEnumerator OnTriggerExit(Collider other) {
   Vector3 spawnPosition = new Vector3 (spawnValues.x, spawnValues.y, spawnValues.z);
   Instantiate (pong, spawnPosition, transform.rotation);
   yield return spawnWait;
 }

 IEnumerator Spawning(){
   yield return new WaitForSeconds (startWait);
   for (int i = 0; i < pongCount; i++) {
    Vector3 spawnPosition = new Vector3 (spawnValues.x, spawnValues.y, spawnValues.z);
    Instantiate (pong, spawnPosition, transform.rotation);
    yield return new WaitForSeconds (spawnWait);
   }
   yield return new WaitForSeconds(spawnWait);
 }
}
